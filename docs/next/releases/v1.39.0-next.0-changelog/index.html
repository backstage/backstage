      useEntityList,
    } from '@backstage/plugin-catalog-react';
    // ...
    const { updateFilters } = useEntityList();

    // ...
    updateFilters({
      order: new EntityOrderFilter([
        {
          field: 'metadata.name',
          order: 'desc',
